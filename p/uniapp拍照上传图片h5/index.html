createInputAndSetAttribute() {
    var input = document.createElement('input');
    input.type = 'file';
    input.id = 'take-picture'
    input.accept = 'image/*';
    // input.capture = 'user';
    input.style.opacity = 0;
    input.style.position = 'absolute';
    input.style.top = 0;
    input.style.left = 0;
    input.style.width = '50px';
    input.style.height = '50px';
    return input;
}
